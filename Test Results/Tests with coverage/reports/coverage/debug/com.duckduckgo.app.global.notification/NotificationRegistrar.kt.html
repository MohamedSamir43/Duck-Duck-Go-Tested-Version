package com.duckduckgo.app.global.notification

import android.annotation.TargetApi
import android.app.NotificationChannel
import android.app.NotificationManager
import android.os.Build
import timber.log.Timber
import javax.inject.Inject

class NotificationRegistrar @Inject constructor(private val notificationManager: NotificationManager) {

    fun registerApp() {
        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.O) {
            Timber.d("No need to register for notification channels on this SDK version")
            return
        }
        configureNotificationChannels()
    }

    @TargetApi(Build.VERSION_CODES.O)
    private fun configureNotificationChannels() {
        val importance = NotificationManager.IMPORTANCE_LOW

        // downloading notification
        val fileDownloadingChannel = NotificationChannel(FILE_DOWNLOADING_CHANNEL_ID, FILE_DOWNLOAD_CHANNEL_NAME, importance)

        // downloaded notification
        val fileDownloadedChannel = NotificationChannel(FILE_DOWNLOADED_CHANNEL_ID, FILE_DOWNLOAD_CHANNEL_NAME, importance)
        notificationManager.createNotificationChannels(
            listOf(
                fileDownloadedChannel,
                fileDownloadingChannel
            )
        )
    }

    companion object {
        const val FILE_DOWNLOADING_CHANNEL_ID = "DOWNLOADING"
        const val FILE_DOWNLOADED_CHANNEL_ID = "DOWNLOADED"
        const val FILE_DOWNLOAD_CHANNEL_NAME = "File Download"
    }
}
